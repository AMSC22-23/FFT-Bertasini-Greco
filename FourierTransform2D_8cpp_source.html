105 void FourierTransform2D<FT>::OutputSpace::pass_filter(const double cutoff_perc, const bool is_high_pass)
106 {

108  auto rows = data[0].size();
109  auto cols = data[0][0].size();
110 
111  unsigned long int area_square = rows * cols;
112  // r = √((g/100) × x^2 / π)
113  auto cutoff_freq = sqrt((1 - cutoff_perc) * (double)area_square / M_PI);
114 
115  pair<int, int> center = {rows / 2, cols / 2};
116  // lamda for distance calculation
117  auto distance = [&](int i, int j) { return sqrt(pow(i - center.first, 2) + pow(j - center.second, 2)); };
118

120  for (auto c = 0ull; c < channels; ++c)
121  for (auto i = 0ull; i < rows; ++i)
122  for (auto j = 0ull; j < cols; ++j)
123  if ((is_high_pass && distance(i,j) > cutoff_freq) || (!is_high_pass && distance(i,j) <= cutoff_freq))
124  data[c][i][j] = 0.0;
